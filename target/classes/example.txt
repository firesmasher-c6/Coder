# This is Coder Plugin
# Coder Plugin Introduces Phyton And Java execution
# This is example.txt

# To add scripts you must use:
.py - for phyton
.java for java
.cpp for c++
.c for c 
.lua for lua
# Add these your script files at YOURSERVER/plugins/Coder/scripts/

# Commands:
/coder run <name> - Runs the script(phyton,java,c,c++, or lua)
/coder reload <name> - reloads the script.

<---- EXAMPLES ---->

== PHYTON ==
print("HELLO WORLD!")

== JAVA ==
public class HelloWorld {
    public static void main(String[] args) {
        System.out.println("Hello, World!");
    }
}

== C ==
#include <stdio.h>

int main() {
    printf("Hello, World!\n");
    return 0;
}

== C++ ==
#include <iostream>

int main() {
    std::cout << "Hello World!";
    return 0;
}

== LUA ==
print("Hello World!")
